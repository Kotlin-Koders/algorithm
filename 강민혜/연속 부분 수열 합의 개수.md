# 연속 부분 수열 합의 개수

20분

## 풀이

```js
function solution(elements) {
  const doubleArr = [...elements, ...elements];
  const len = elements.length;

  const set = new Set();

  for (let i = 1; i <= len; i++) {
    for (let j = 0; j < len; j++) {
      const sum = doubleArr.slice(j, j + i).reduce((acc, cur) => acc + cur);
      set.add(sum);
    }
  }

  return [...set].length;
}
```

```js
function solution(elements) {
  const doubleArr = [...elements, ...elements];
  const len = elements.length;

  const set = new Set();

  for (let i = 1; i <= len; i++) {
    for (let j = 0; j < len; j++) {
      let sum = 0;
      for (let k = j; k < j + i; k++) {
        sum += doubleArr[k];
      }
      set.add(sum);
    }
  }

  return [...set].length;
}
```
