import java.util.*
class Solution {
    fun solution(operations: Array<String>): IntArray {
        val asc = PriorityQueue<Int>()
        val desc = PriorityQueue<Int>(Comparator.reverseOrder())
        var minCount = 0
        var maxCount = 0
        for (operation in operations) {
            if (operation == "D -1") {
                if (!asc.isEmpty() && asc.size > maxCount) {
                    asc.poll()
                }
                minCount += 1
            } else if (operation == "D 1") {
                if (!desc.isEmpty() && desc.size > minCount) {
                    desc.poll()
                }
                maxCount += 1
            } else {
                val n = (operation.split(" ")[1]).toInt()
                asc.add(n)
                desc.add(n)
            }
        }
        
        if (asc.size <= minCount + maxCount) {
            return intArrayOf(0,0)
        }
    
        return intArrayOf(desc.peek(), asc.peek())
    }
}
