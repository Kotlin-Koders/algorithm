# 리코쳇 로봇

```kotlin
import java.util.*

class Solution {
    fun solution(board: Array<String>): Int {
        var answer: Int = 0
        val directions = arrayOf(
            Pair(-1, 0), // 상
            Pair(1, 0),  // 하
            Pair(0, -1), // 좌
            Pair(0, 1)   // 우
        )
        var visited = mutableListOf<String>()

        var start = Pair(Pair(0, 0), 0)

        for ((r, s) in board.withIndex()) {
            for ((c, ch) in s.withIndex()) {
                if (ch == 'R') {
                    start = Pair(Pair(r, c), 0)
                }
            }
        }

        var stack = ArrayDeque<Pair<Pair<Int, Int>, Int>>()
        stack.addFirst(start)

        while (stack.isNotEmpty()) {
            val now = stack.removeFirst()
            val row = now.first.first
            val column = now.first.second
            visited.add("${row}-${column}")

            if (board[row][column] == 'G') {
                return now.second
            }

            for (dir in directions) {
                var pos = Pair(row, column)
                while (true) {
                    var movePos = Pair(pos.first + dir.first, pos.second + dir.second)
                    if (0 <= movePos.first && movePos.first < board[0].length &&
                        0 <= movePos.second && movePos.second < board.size) {

                        if (visited.contains("${movePos.first}-${movePos.second}")) {
                            break
                        }

                        if (board[movePos.first][movePos.second] == 'D') {
                            stack.addFirst(Pair(pos, now.second + 1))
                            break
                        } else {
                            pos = movePos
                        }
                    } else if (!visited.contains("${movePos.first}-${movePos.second}")) {
                        stack.addFirst(Pair(pos, now.second + 1))
                        break
                    }
                }
            }
        }
        print(visited)
        return answer
    }
}
```
